```python
def calcular_pendiente(x1, y1, x2, y2):

    # Verifica si los puntos tienen la misma coordenada x para evitar divisi칩n por cero
    if x1 == x2:
        return "La pendiente es indefinida (l칤nea vertical)"

    return round(pendiente,2)
```

```python
    pares = 0
    impares = 0

    print(f'La cantidad de numeros impares es {impares}')
```

```python
def factorial(numero):
    # Inicializamos total a 1 porque el factorial se calcula con multiplicaciones
    total = 1
    # Recorremos todos los n칰meros de 1 a 'numero' (inclusive)
    for num in range(1, numero + 1):
        total *= num  # Multiplicamos 'total' por cada n칰mero en el rango
    return total  # Devolvemos el resultado final

# Ejemplo de uso
print(factorial(5))
```
